pipeline {
    agent any

    environment {
        SQLSERVER = "DEVELOPER1"
        DATABASE  = "ErrorDB"
        CSV_DIR   = 'csvfile'
    }

    stages {
        stage('checkout code from github') {
            steps {
                checkout scm
            }
        }

        stage('Inserting csv into sql server') {
            steps {
                script {
                    def files = findFiles(glob: "${env.CSV_DIR}/*.csv")
                    if (files.length == 0) {
                        echo "No file found in ${env.CSV_DIR}"
                        error "No files to process"
                    }

                    for (file in files) {
                        def filepath = file.path.replace("/", "\\")
                        echo "Processing csv file: ${filepath}"

                        //  Define the bulk insert query BEFORE using it
                        def bulkInsertQuery = """
                        BULK INSERT ErrorBD.dbo.population
                        FROM '${filepath}'
                        WITH (
                            FIRSTROW = 2,
                            FIELDTERMINATOR = ',',
                            ROWTERMINATOR = '\\n',
                            TABLOCK
                        );
                        """

                        echo "Executing query: ${bulkInsertQuery}"

                        // Run SQLCMD command
                        bat """
                        sqlcmd -S ${env.SQLSERVER} -d ${env.DATABASE} -E -Q "${bulkInsertQuery}"
                        """
                    }
                }
            }
        }
    }
}
